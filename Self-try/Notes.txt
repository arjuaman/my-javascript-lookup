function move(e){
    if(e.keyCode == 39 || e.keyCode == 68)
        shapeBoundingBox.left+=speed;
    if(e.keyCode == 37 || e.keyCode == 65)
        shapeBoundingBox.left-=speed;
    if(e.keyCode == 38 || e.keyCode == 87)
        shapeBoundingBox.top-=speed;
    if(e.keyCode == 40 || e.keyCode == 83)
        shapeBoundingBox.top+=speed;
}
